    <script>
        document.addEventListener('DOMContentLoaded', async () => {
            const authToken = localStorage.getItem('authToken');
            if (!authToken) {
                window.location.href = 'index.html';
                return;
            }

            const modal = document.getElementById('subscriptionModal');
            const modalClose = document.getElementById('modalClose');
            const modalCancel = document.getElementById('modalCancel');
            const modalConfirm = document.getElementById('modalConfirm');
            const groupSelect = document.getElementById('groupSelect');
            const subscriptionType = document.getElementById('subscriptionType');
            const availableSeats = document.getElementById('availableSeats');
            const notification = document.getElementById('notification');

            let currentGroupId = '';
            let allGroups = [];
            let userGroups = [];
            let userSubscriptions = [];

            function showNotification(message, isError = false) {
                notification.textContent = message;
                notification.style.background = isError ? '#FF6B6B' : '#FA98C8';
                notification.classList.add('show');
                setTimeout(() => notification.classList.remove('show'), 3000);
            }

            function openModal(initialType = '8') {
                subscriptionType.value = initialType;
                updateGroupSelect();
                modal.classList.add('active');
            }

            function closeModal() {
                modal.classList.remove('active');
            }

            function updateGroupSelect() {
                groupSelect.innerHTML = '<option value="">-- Выберите группу --</option>';
                allGroups.forEach(group => {
                    if (!userGroups.some(g => g.id === group.id)) {
                        const name = group.Наименование || group.name || 'Без названия';
                        const instructor = group.Преподаватель || group.instructor || 'Без преподавателя';
                        const seats = parseInt(group.МаксимальноеКоличествоМест) || group.maxSeats || 0;

                        const option = document.createElement('option');
                        option.value = group.id;
                        option.textContent = `${name} (${instructor}) — ${seats} мест`;
                        option.dataset.seats = seats;
                        groupSelect.appendChild(option);
                    }
                });
            }

            modalClose.addEventListener('click', closeModal);
            modalCancel.addEventListener('click', closeModal);

            groupSelect.addEventListener('change', () => {
                const selectedOption = groupSelect.options[groupSelect.selectedIndex];
                currentGroupId = groupSelect.value;
                availableSeats.textContent = selectedOption ? `${selectedOption.dataset.seats} мест` : '';
            });

            modalConfirm.addEventListener('click', async () => {
                const groupId = groupSelect.value;
                const type = subscriptionType.value;

                if (!groupId || !type) {
                    showNotification('Выберите группу и тип абонемента', true);
                    return;
                }

                try {
                    // 1. Сначала записываем в группу
                    const joinResponse = await fetch('/api/join-group', {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                            'Authorization': `Bearer ${authToken}`
                        },
                        body: JSON.stringify({ groupId })
                    });

                    if (!joinResponse.ok) {
                        const error = await joinResponse.json();
                        throw new Error(error.error || 'Ошибка записи в группу');
                    }

                    // 2. Теперь покупаем абонемент
                    const subscribeResponse = await fetch('/api/subscribe', {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                            'Authorization': `Bearer ${authToken}`
                        },
                        body: JSON.stringify({ subscriptionType: type, groupId })
                    });

                    if (!subscribeResponse.ok) {
                        const error = await subscribeResponse.json();
                        throw new Error(error.error || 'Ошибка покупки абонемента');
                    }

                    // 3. Обновляем интерфейс
                    const updatedGroups = allGroups.map(group =>
                        group.id === groupId
                            ? { ...group, maxSeats: group.maxSeats - 1 }
                            : group
                    );
                    allGroups = updatedGroups;

                    showNotification('Абонемент успешно куплен!');
                    closeModal();
                    await loadUserData(); // Перезагружаем данные пользователя

                } catch (error) {
                    showNotification(error.message, true);
                }
            });

            function updateAvailableGroups() {
                const container = document.getElementById('availableGroups');
                const available = allGroups.filter(group => !userGroups.some(g => g.id === group.id));

                if (available.length) {
                    container.innerHTML = '';
                    available.forEach(group => {
                        const div = document.createElement('div');
                        div.className = 'group-card';
                        div.innerHTML = `
                                <h3>${group.Наименование || group.name || 'Название не указано'}</h3>
                                <p>Направление: ${group.Направление || group.direction || 'Не указано'}</p>
                                <p>Преподаватель: ${group.Преподаватель || group.instructor || 'Не указан'}</p>
                                <p>Расписание: ${group.Расписание || group.schedule || 'Не указано'}</p>
                                <p>Адрес: ${group.Адрес || group.address || 'Не указан'}</p>
                                <p>Свободных мест: ${parseInt(group.МаксимальноеКоличествоМест) || group.maxSeats || 0}</p>
                                <button class="join-button" data-group-id="${group.id}">Записаться</button>
                            `;

                        container.appendChild(div);
                    });

                    document.querySelectorAll('.join-button').forEach(button => {
                        button.addEventListener('click', () => {
                            const groupId = button.getAttribute('data-group-id');
                            const group = allGroups.find(g => g.id === groupId);

                            if (!group) return;

                            // Установим выбранную группу в выпадающий список
                            openModal(); // сначала открыть модалку
                            updateGroupSelect(); // обновить список групп

                            setTimeout(() => {
                                groupSelect.value = groupId;
                                availableSeats.textContent = `${parseInt(group.МаксимальноеКоличествоМест) || group.maxSeats || 0} мест`;
                            }, 0);
                        });
                    });
                } else {
                    container.innerHTML = '<p>Нет доступных групп для записи.</p>';
                }
            }

            function displayUserGroupsAndSubscriptions() {
                const container = document.getElementById('userGroups');
                if (!userGroups.length && !userSubscriptions.length) {
                    container.innerHTML = '<p>Вы пока не записаны в группы.</p>';
                    return;
                }

                container.innerHTML = '';
                const grouped = {};
                userSubscriptions.forEach(sub => {
                    if (!grouped[sub.groupId]) grouped[sub.groupId] = [];
                    grouped[sub.groupId].push(sub);
                });

                userGroups.forEach(group => {
                    const div = document.createElement('div');
                    div.className = 'group-card';
                    const subs = grouped[group.id] || [];
                    const total = subs.reduce((sum, s) => sum + (s.remaining || 0), 0);

                    div.innerHTML = `
                            <h3>${group.name || 'Группа'}</h3>
                            <p>Направление: ${group.direction || 'Не указано'}</p>
                            <p>Преподаватель: ${group.instructor || 'Не указан'}</p>
                            <p>Расписание: ${group.schedule || 'Не указано'}</p>
                            <p>Адрес: ${group.address || 'Не указан'}</p>
                            ${subs.length ? `
                                <div class="subscription-item">
                                    <div class="subscription-details">
                                        <h4>Ваши абонементы</h4>
                                        <span class="subscription-status status-active">Активен</span>
                                    </div>
                                    <p>Осталось занятий: ${total}</p>
                                </div>
                            ` : ''}
                        `;
                    container.appendChild(div);
                });
            }

            async function loadUserData() {
                try {
                    const userRes = await fetch('/api/user', {
                        headers: { 'Authorization': `Bearer ${authToken}` }
                    });

                    if (!userRes.ok) {
                        localStorage.removeItem('authToken');
                        window.location.href = 'index.html';
                        return;
                    }

                    const userData = await userRes.json();
                    document.getElementById('userName').textContent = userData.name;
                    document.getElementById('userEmail').textContent = userData.email;
                    document.getElementById('userPhone').textContent = userData.phone;
                    userGroups = userData.groups || [];
                    userSubscriptions = userData.subscriptions || [];
                    displayUserGroupsAndSubscriptions();
