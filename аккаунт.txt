    <script>
        document.addEventListener('DOMContentLoaded', async () => {
            const authToken = localStorage.getItem('authToken');
            if (!authToken) {
                window.location.href = 'index.html';
                return;
            }

            // Инициализация модального окна
            const modal = document.getElementById('subscriptionModal');
            const modalClose = document.getElementById('modalClose');
            const modalCancel = document.getElementById('modalCancel');
            const modalConfirm = document.getElementById('modalConfirm');
            const groupSelect = document.getElementById('groupSelect');
            const subscriptionType = document.getElementById('subscriptionType');
            const availableSeats = document.getElementById('availableSeats');
            const notification = document.getElementById('notification');

            let currentGroupId = '';
            let allGroups = [];
            let userGroups = [];
            let userSubscriptions = [];

            // Функция для показа уведомлений
            function showNotification(message, isError = false) {
                notification.textContent = message;
                notification.style.background = isError ? '#FF6B6B' : '#FA98C8';
                notification.classList.add('show');

                setTimeout(() => {
                    notification.classList.remove('show');
                }, 3000);
            }

            // Функции для работы с модальным окном
            function openModal(initialType = '8') {
                subscriptionType.value = initialType;
                updateGroupSelect();
                modal.classList.add('active');
            }

            function closeModal() {
                modal.classList.remove('active');
            }

            function updateGroupSelect() {
                groupSelect.innerHTML = '<option value="">-- Выберите группу --</option>';
                allGroups.forEach(group => {
                    if (!userGroups.some(g => g.Ссылка === group.id)) {
                        const option = document.createElement('option');
                        option.value = group.id;
                        option.textContent = `${group.Наименование} (${group.Преподаватель}) - ${group.МаксимальноеКоличествоМест} мест`;
                        option.dataset.seats = group.МаксимальноеКоличествоМест;
                        groupSelect.appendChild(option);
                    }
                });
            }

            // Обработчики событий модального окна
            modalClose.addEventListener('click', closeModal);
            modalCancel.addEventListener('click', closeModal);

            groupSelect.addEventListener('change', () => {
                currentGroupId = groupSelect.value;
                const selectedOption = groupSelect.options[groupSelect.selectedIndex];
                availableSeats.textContent = selectedOption ? `${selectedOption.dataset.seats} мест` : '';
            });

            modalConfirm.addEventListener('click', async () => {
                const groupId = groupSelect.value;
                const type = subscriptionType.value;

                if (!groupId || !type) {
                    showNotification('Пожалуйста, выберите группу и тип абонемента', true);
                    return;
                }

                try {
                    console.log('Отправка запроса:', { subscriptionType: type, groupId });

                    const response = await fetch('/api/subscribe', {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                            'Authorization': `Bearer ${authToken}`
                        },
                        body: JSON.stringify({
                            subscriptionType: type,
                            groupId: groupId
                        })
                    });

                    if (!response.ok) {
                        const errorData = await response.json();
                        console.error('Ошибка сервера:', errorData);
                        throw new Error(errorData.error || 'Неизвестная ошибка сервера');
                    }

                    const result = await response.json();
                    console.log('Успешный ответ:', result);

                    showNotification('Абонемент успешно приобретен!');

                    // Обновляем количество мест в группе
                    const groupToUpdate = allGroups.find(g => g.Ссылка === groupId);
                    if (groupToUpdate) {
                        groupToUpdate.МаксимальноеКоличествоМест -= 1;
                    }

                    // Обновляем отображение групп
                    updateAvailableGroups();

                    closeModal();
                    await loadUserData(); // Перезагружаем данные пользователя
                } catch (error) {
                    console.error('Ошибка при покупке абонемента:', error);
                    showNotification(error.message || 'Ошибка соединения с сервером', true);
                }
            });

            // Обновление списка доступных групп
            function updateAvailableGroups() {
                const availableGroupsContainer = document.getElementById('availableGroups');
                const availableGroups = allGroups.filter(
                    group => !userGroups.some(g => g.Ссылка === group.id)
                );

                if (availableGroups.length > 0) {
                    availableGroupsContainer.innerHTML = '';
                    availableGroups.forEach(group => {
                        const groupElement = document.createElement('div');
                        groupElement.className = 'group-card';
                        groupElement.innerHTML = `
                                    <h3>${group.Наименование}</h3>
                                    <p>Направление: ${group.Направление}</p>
                                    <p>Преподаватель: ${group.Преподаватель}</p>
                                    <p>Расписание: ${group.Расписание}</p>
                                    <p>Адрес: ${group.Адрес}</p>
                                    <p>Свободных мест: ${group.МаксимальноеКоличествоМест}</p>
                                    <button class="join-button" data-group-id="${group.id}">Записаться</button>
                                `;
                        availableGroupsContainer.appendChild(groupElement);
                    });

                    // Обработка кнопки записи в группу
                    document.querySelectorAll('.join-button').forEach(button => {
                        button.addEventListener('click', async () => {
                            const groupId = button.getAttribute('data-group-id');

                            try {
                                const response = await fetch('/api/join-group', {
                                    method: 'POST',
                                    headers: {
                                        'Content-Type': 'application/json',
                                        'Authorization': `Bearer ${authToken}`
                                    },
                                    body: JSON.stringify({
                                        groupId: groupId
                                    })
                                });

                                const result = await response.json();

                                if (result.success) {
                                    showNotification('Вы успешно записаны в группу!');
                                    await loadUserData(); // Перезагружаем данные
                                } else {
                                    showNotification(result.error || 'Ошибка при записи в группу', true);
                                }
                            } catch (error) {
                                showNotification('Ошибка соединения с сервером', true);
                            }
                        });
                    });
                } else {
                    availableGroupsContainer.innerHTML = '<p>Нет доступных групп для записи.</p>';
                }
            }

            // Отображение групп и абонементов пользователя
            function displayUserGroupsAndSubscriptions() {
                const container = document.getElementById('userGroups');

                if (userGroups.length === 0 && userSubscriptions.length === 0) {
                    container.innerHTML = '<p>Вы пока не записаны в группы.</p>';
                    return;
                }

                container.innerHTML = '';

                // Группируем абонементы по группам
                const groupedSubscriptions = {};
                userSubscriptions.forEach(sub => {
                    if (!groupedSubscriptions[sub.groupId]) {
                        groupedSubscriptions[sub.groupId] = [];
                    }
                    groupedSubscriptions[sub.groupId].push(sub);
                });

                // Отображаем группы с абонементами
                userGroups.forEach(group => {
                    const groupElement = document.createElement('div');
                    groupElement.className = 'group-card';

                    // Находим абонементы для этой группы
                    const subscriptions = groupedSubscriptions[group.id] || [];
                    const totalRemaining = subscriptions.reduce((sum, sub) => sum + sub.remaining, 0);

                    let subscriptionsHTML = '';
                    if (subscriptions.length > 0) {
                        subscriptionsHTML = `
                                    <div class="subscription-item">
                                        <div class="subscription-details">
                                            <h4>Ваши абонементы</h4>
                                            <span class="subscription-status status-active">Активен</span>
                                        </div>
                                        <p>Осталось занятий: ${totalRemaining}</p>
                                    </div>
                                `;
                    }

                    groupElement.innerHTML = `
                                <h3>${group.Наименование}</h3>
                                <p>Направление: ${group.Направление}</p>
                                <p>Преподаватель: ${group.Преподаватель}</p>
                                <p>Расписание: ${group.Расписание}</p>
                                <p>Адрес: ${group.Адрес}</p>
                                ${subscriptionsHTML}
                            `;

                    container.appendChild(groupElement);
                });
            }

            // Загрузка данных пользователя
            async function loadUserData() {
                try {
                    const userResponse = await fetch('/api/user', {
                        headers: {
                            'Authorization': `Bearer ${authToken}`
                        }
                    });

                    if (!userResponse.ok) {
                        localStorage.removeItem('authToken');
                        window.location.href = 'index.html';
                        return;
                    }

                    const userData = await userResponse.json();
                    console.log('Данные пользователя:', userData);

                    // Заполняем данные пользователя
                    document.getElementById('userName').textContent = userData.name;
                    document.getElementById('userEmail').textContent = userData.email;
                    document.getElementById('userPhone').textContent = userData.phone;

                    // Сохраняем группы и абонементы пользователя
                    userGroups = userData.groups || [];
                    userSubscriptions = userData.subscriptions || [];

                    // Отображаем группы и абонементы пользователя
                    displayUserGroupsAndSubscriptions();

                    // Загрузка доступных групп
                    const groupsResponse = await fetch('/api/groups');
